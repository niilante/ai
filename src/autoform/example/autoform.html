The autoform modules is not meant for production code but rather is a tool for use during development. It enables you to quickly prototype some form action without building a bunch of markup. Similar to what you might have done with .NET editor templates. Be advised this is an alpha not yet even beta module so please use with caution.

  • Code
  • Options
<form name="autoForm" ai-autoform ai-autoform-options="autoOptions" ></form>

$scope.autoSource = {
    name: 'Nancy Brewer',
    email: 'nancy@gmail.com',
    phone: '8885551212',
    age: 23,
    enabled: true
};
$scope.autoOptions = {
    elements: {
        email: { type: 'email' },
        age: { type: 'radio', values: [19,20,21,22,23,24,25] }
    },
    source: 'autoSource'
};

var defaults = {
    prefix: 'model',           // value to prepend to models.
    source: undefined,         // the source data for the form.
    labels: undefined,         // when true labels are created.
    type: 'text',              // the default type for elements.
    addClass: false,           // specify class to be added to form for styling.
    textareaAt: 35,            // if value is greater than this use textarea.

    datetimeExp:               // date time expression
        /^(0?[1-9]|1[012])(:[0-5]\d) [APap][mM]$/,

    phoneExp:                   // phone expression
        /^(?!.*911.*\d{4})((\+?1[\/ ]?)?(?![\(\. -]?555.*)\( ?[2-9][0-9]{2} ?\) ?|(\+?1[\.\/ -])?[2-9][0-9]{2}[\.\/ -]?)(?!555.?01..)([2-9][0-9]{2})[\.\/ -]?([0-9]{4})$/,

    intExp:                     // integer/number expression.
        /^\d+$/,
                                // email expression.
    emailExp:
        /^[\w-]+(\.[\w-]+)*@([a-z0-9-]+(\.[a-z0-9-]+)*?\.[a-z]{2,6}|(\d{1,3}\.){3}\d{1,3})(:\d{4})?$/
}
